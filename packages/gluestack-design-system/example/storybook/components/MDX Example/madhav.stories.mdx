import { Canvas, Meta, Story } from '@storybook/addon-docs';
import { Badge, Center } from '@gluestack/design-system';

<Meta title="MDX/Badge" component={Badge} />

export const Template = ({ variant, text, ...props }) => (
  <Badge variant={variant} {...props}>
    <Badge.Text>{text}</Badge.Text>
    <Badge.Icon sx={{ style: { pl: 4 } }}></Badge.Icon>
  </Badge>
);

# Badge

Let's define a story for our `Badge` component:

<Story
  name="positive"
  args={{
    text: 'New feature',
    variant: 'subtle',
  }}
>
  {Template.bind({})}
</Story>

We can drop it in a `Canvas` to get a code snippet:

<Canvas>
  <Story
    name="negative"
    args={{
      text: 'New feature',
      variant: 'subtle',
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>

We can even preview multiple Stories in a block. This
gets rendered as a group but defines individual stories
with unique URLs, which is great for review and testing.
