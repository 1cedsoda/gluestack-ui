# Installation

- Install library

  ```bash
  # With npm
  npm install @dank-style/react

  # With yarn
  yarn add @dank-style/react
  ```

- Import
  ```jsx
  import { styled } from '@dank-style/react';
  ```
- Usage

  ```jsx
  import { StyledProvider, styled } from '@dank-style/react';

  const StyledView = styled(
    View,
    {
      baseStyle: {
        style: {
          bg: '$red500',
          p: '$3',
        },
      },
    },
    {}
  );

  const App = () => (
    <StyledProvider>
      <StyledView />
    </StyledProvider>
  );
  ```
