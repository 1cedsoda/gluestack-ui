<br />

## API

<br />

```jsx
import React from 'react';
import { View } from 'react-native';
import { styled } from '@dank-style/react';

const StyledButtonStateProps = styled(
  Pressable,
  {
    baseStyle: {
      style: {
        bg: '$primary600',
        p: '$3',
      },
      descendants: {
        _text: {
          style: {
            color: '$white',
          },
        },
      },
      state: {
        hover: {
          style: {
            bg: '$primary700',
          },
        },
      },
    },
  },
  { descendantStyle: ['_text'] }
);

const StyledButtonText = styled(
  Text,
  {},
  {
    ancestorStyle: ['_text'],
  }
);
```
