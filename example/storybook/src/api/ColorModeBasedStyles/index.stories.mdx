# ColorMode Based Styles

<br />
<br />
<br />

## API

<br />

```jsx
import React from 'react';
import { View } from 'react-native';
import { styled } from '@dank-style/react';

const StyledColorMode = styled(
  View,
  {
    baseStyle: {
      style: {
        w: 100,
        h: 100,
        bg: '$red500',
      },
      colorMode: {
        dark: {
          style: {
            bg: '$info600',
          },
        },
      },
    },
  },
  {}
);
```
